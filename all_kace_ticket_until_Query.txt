select HD_TICKET.ID, 
       HD_TICKET.TITLE,
       HD_TICKET.CUSTOM_FIELD_VALUE0 as Dept,
       HD_QUEUE.NAME as QUEUE,
       HD_PRIORITY.NAME as PRIORITY,
       HD_CATEGORY.NAME as CATEGORY,
       HD_STATUS.NAME as STATUS,
       HD_IMPACT.NAME as IMPACT,
       HD_TICKET.CREATED,
       HD_TICKET.TIME_OPENED,
       HD_TICKET.TIME_CLOSED,
       HD_TICKET.SATISFACTION_RATING
       FROM 
ORG1.HD_TICKET
inner join HD_QUEUE ON HD_QUEUE_ID = HD_QUEUE.ID
left join HD_STATUS on HD_STATUS_ID = HD_STATUS.ID
left join HD_PRIORITY on HD_PRIORITY_ID = HD_PRIORITY.ID
left join HD_CATEGORY ON HD_CATEGORY_ID = HD_CATEGORY.ID
left join HD_IMPACT ON HD_IMPACT_ID = HD_IMPACT.ID
#INNER join HD_QUEUE ON HD_QUEUE_ID = HD_QUEUE.ID

ORDER BY HD_TICKET.CREATED ASC
